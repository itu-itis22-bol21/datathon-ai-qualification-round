import os
import pandas as pd
from PIL import Image
from torch.utils.data import Dataset

class CustomDataset(Dataset):
    def __init__(self, csv_file, root_dir, transform=None):
        """
        Args:
            csv_file (str): Path to the CSV file with image paths and labels.
            root_dir (str): Directory with all the images.
            transform (callable, optional): Optional transform to be applied on an image.
        """
        self.data = pd.read_csv(csv_file)  # Load the CSV file
        self.root_dir = root_dir
        self.transform = transform
        self.classes = sorted(self.data['city'].unique())  # Extract class labels
        self.class_to_idx = {cls: idx for idx, cls in enumerate(self.classes)}  # Map classes to indices

    def __len__(self):
        return len(self.data)

    def __getitem__(self, idx):
        img_name = os.path.join(self.root_dir, self.data.iloc[idx, 0])  # Image path
        label = self.data.iloc[idx, 1]  # Corresponding label
        label_idx = self.class_to_idx[label]  # Convert label to class index
        
        image = Image.open(img_name).convert('RGB')  # Open the image
        
        if self.transform:
            image = self.transform(image)  # Apply transformations
        
        return image, label_idx

from torchvision import transforms

# Define transforms for data augmentation/preprocessing
mean = [0.4363, 0.4328, 0.3291]
std = [0.2129, 0.2075, 0.2038]

train_transforms = transforms.Compose([
    transforms.Resize((224,224)),#Resizing images to make faster ML training
    transforms.RandomHorizontalFlip(),
    transforms.RandomRotation(10),
    transforms.ToTensor(),
    transforms.Normalize(torch.Tensor(mean), torch.Tensor(std)),
])

test_transforms = transforms.Compose([
    transforms.Resize((224,224)),#Resizing images to make faster ML training
    transforms.ToTensor(),
    transforms.Normalize(torch.Tensor(mean), torch.Tensor(std)),
])


train_csv = 'train/train_data.csv'  # Path to your training CSV
test_csv = 'test/test.csv'  # Path to your testing CSV
train_dir = 'train/train/'  # Directory containing training images
test_dir = 'test/test/'  # Directory containing test images

# Create dataset instances
train_dataset = CustomDataset(csv_file=train_csv, root_dir=train_dir, transform=train_transforms)
test_dataset = CustomDataset(csv_file=test_csv, root_dir=test_dir, transform=test_transforms)
